# StateHook.jsx

import React, { useState } from 'react';

function StateHook() {
  // 声明一个叫 "count" 的 state 变量
  const [obj, setCountFn] = useState({a:{b:{c:0}}});

  const clickFn=()=>{
    obj.a.b.c+=1
    // 需要得到一个obj的副本才能触发更新
    const newObj=JSON.parse(JSON.stringify(obj))
    setCountFn(newObj) 
  }

  // function clickFn()  {
  //   obj.a.b.c+=1
  //   // 需要得到一个obj的副本才能触发更新
  //   const newObj=JSON.parse(JSON.stringify(obj))
  //   setCountFn(newObj) 
  // }

  return (
    <div>
      <p>You clicked {obj.a.b.c} times</p>
      <button onClick={clickFn}>
        Click me
      </button>
    </div>
  );
}

export default StateHook

# EffectHook.jsx

import React, { useState, useEffect } from 'react';

export default () => {
  const [count, setCount] = useState(0);
  const [name, setName] = useState(0);

  useEffect(() => {
    document.title = `You clicked ${count} times`;
    
    // 清除函数  相当于componentWillUnmount
    // return function clearFn(){
    //   console.log('update 会触发清除')
    // }
    return ()=>{
      console.log('update 会触发清除')
    }
  },[count]);

  /**
   *  第二个参数
   *  [] 等价于只在componentDidMount的时候执行
   *  [count] 只监听count改变的时候才会触发
   *  没有第二个参数 等价于在componentDidMount,componentDidUpdate的时候执行
   */
  useEffect(() => {
    console.log('多个 useEffect')

    return ()=>{  // 第二个参数传[]会失效
      console.log('第二个参数测试')
    }
  },[name]);

  return (
    <div>
      <p>You clicked {count} times</p>
      <p>{name}</p>
      <button onClick={() => setCount(count + 1)}>
        Click me
      </button>
      <button onClick={() => setName('user'+Math.random()*10)}>
        change name
      </button>
    </div>
  );
}

# UseMemoHook.jsx

import React, { useState, useMemo } from 'react';

export default () => {
  const [count, setCount] = useState(0)
  const [otherVar,setOtherVar] = useState(100)

  const useMemoResult = useMemo(changeName,[count])

  function changeName(){ // 跟 vue 的computed很相似 只不过自定义了依赖
    let str=`otherVar:${otherVar} count:${count}`
    return str
  }

  return (
    <div>
      <p>You clicked {count} times</p>
      <button onClick={() => setCount(count + 1)}>
        Click me
      </button>

      <p>{useMemoResult}</p>

      <p>{otherVar}</p>
      <button onClick={()=>setOtherVar(otherVar+1)}>
        change otherVar
      </button>
    </div>
  );
}

# UseCallbackHook.jsx

import React, { useState, useCallback } from 'react';

const Button = React.memo(({ onClickButton, children }) => {
  return (
    <>
      <button onClick={onClickButton}>{children}</button>
      <span>{Math.random()}</span>
    </>
  );
});


export default function App() {
  const [count1, setCount1] = useState(0);
  const [count2, setCount2] = useState(0);

  const handleClickButton1 = () => {
    setCount1(count1 + 1);
  };

  // 依赖不变 函数引用不变
  const handleClickButton2 = useCallback(() => {
    setCount2(count2 + 1);
  }, [count2]);

  return (
    <div>
      <div>
        <Button onClickButton={handleClickButton1}>Button1</Button>
      </div>
      <div>
        <Button onClickButton={handleClickButton2}>Button2</Button>
      </div>
    </div>
  );
}

# ContextHook.jsx

import React,{useContext, useState} from 'react'

const themes = {
  light: {
    foreground: "#000000",
    background: "#eeeeee"
  },
  dark: {
    foreground: "#ffffff",
    background: "#222222"
  }
};

const ThemeContext = React.createContext(themes );

function ContextHook(props) {
  return (
    <ThemeContext.Provider value={themes}>
      {props.children}
    </ThemeContext.Provider>
  );
}

function Toolbar(props) {
  return (
    <div>
      <ThemedButton />
    </div>
  );
}
let i=0
function ThemedButton() {
  const theme = useContext(ThemeContext);
  const [themeStyle,setThemeStyle]=useState(theme.light)
  function changeThem(){
    i++
    setThemeStyle(i%2===1?theme.dark:theme.light)
  }

  return (
    <button style={themeStyle} onClick={changeThem}>
      I am styled by theme context!
    </button>
  );
}

export default (()=>{
  return (<ContextHook>
    <Toolbar/>
  </ContextHook>)
})

# ReducerHook.jsx

import React, {useReducer}from 'react'

const initialState = {count: 0};

function reducer(state, action) {
  switch (action.type) {
    case 'increment':
      return {count: state.count + 1};
    case 'decrement':
      return {count: state.count - 1};
    default:
      throw new Error();
  }
}

function Counter() {
  const [state, dispatch] = useReducer(reducer, initialState);
  return (
    <>
      Count: {state.count}
      <button onClick={() => dispatch({type: 'decrement'})}>-</button>
      <button onClick={() => dispatch({type: 'increment'})}>+</button>
    </>
  );
}

export default Counter

# useContext+useReducer.jsx

import React, { createContext, useReducer, useContext } from "react";

// 创建 context
const ColorContext = createContext({});

// reducer
export const UPDATE_COLOR = "UPDATE_COLOR"
const reducer = (state, action) => {
  switch(action.type) {
    case UPDATE_COLOR:
      return action.color
    default:
      return state  
  }
}


/**
 * 创建一个 Color 组件
 * Color 组件包裹的所有组件都可以访问到 value
 */
export const Color = props => {
  const [color, dispatch] = useReducer(reducer, 'blue')
  console.log(color)
  return (
    <ColorContext.Provider value={{color, dispatch}}>
      {props.children}
    </ColorContext.Provider>
  );
};


const ShowArea = props => {
  const { color } = useContext(ColorContext);
  return <div style={{ color: color }}>字体颜色展示为{color}</div>;
};


const Buttons = props => {
  const { dispatch } = useContext(ColorContext);
  return (
    <React.Fragment>
      <button
        onClick={() => {
          dispatch({ type: UPDATE_COLOR, color: "red" });
        }}
      >
        红色
      </button>
      <button
        onClick={() => {
          dispatch({ type: UPDATE_COLOR, color: "yellow" });
        }}
      >
        黄色
      </button>
    </React.Fragment>
  );
};


export default (()=>{
  return (
    <Color>
      <ShowArea/>
      <Buttons/>
    </Color>
  )
})

import React, { useImperativeHandle, forwardRef } from 'react'
import { useEffect } from 'react';
import { useRef } from 'react';
function FancyInput(props, ref) {
  const inputRef = useRef();
  useImperativeHandle(ref, () => ({
    focus: () => {
      inputRef.current.focus();
    }
  }));
  return (<input ref={inputRef} />);
}
FancyInput = forwardRef(FancyInput);

}

# TodoListDemo.jsx

import React,{useState}  from 'react'

const useInputValue=(initialValue)=>{
  const [value,setValue]=useState(initialValue)

  return {
    value,
    setValue,
    onChange:e=>setValue(e.target.value)
  }
}

const TodoForm=(props)=>{
  const {setValue,...textInputProps}=useInputValue('') // useInputValue 相当于一个自定义 Hook

  function handlerSubmit(e){
    e.preventDefault()
    props.onSubmit(textInputProps.value)
    setValue('')
  }


  return(
    <form onSubmit={handlerSubmit}>
      <input type="text" {...textInputProps}/>
    </form>
  )
}

const TodoList=()=>{
  const [todos,setTodos]=useState([])

  function setTodoList(text){
    setTodos([text,...todos])
  }

  return(
    <div>
      <TodoForm onSubmit={setTodoList}/>
      <ul>
        {
          todos.map((item,index)=>(<li key={index}>{item}</li>) )
        }
      </ul>
    </div>
  )
}

export default TodoList
